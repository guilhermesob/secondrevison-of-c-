//LECTURE : Types in C++


#include <stdio.h>

int main(){
/*
binary  - Uses the prefix 0b
octal - Uses the prefix 0
decimal - Default
hexadecimal - Uses the prefix 0x
*/

unsigned short a = 0b10101010;
printf("%hu\n",a);

int b = 0123;
printf("%d\n",b);

unsigned long long d = 0xFFFFFFFFFFFFFFF;
printf("%llu\n", d);
    return 0;
}
